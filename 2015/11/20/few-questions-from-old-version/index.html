Probelam I: Grey Code

Given 2 numbers. Find if they are consecutive gray (grey) code sequences. In grey code sequence, two succesive values differ in only one bit.

Solution

public class Solution{
	public int greyCode(byte term1, byte term2) {
		// use XOR to find the difference between two numbers
		byte x = (byte)(term1 ^ term2);
		// check how many 1s left in XOR result.
		int total = 0;
		while(x != 0){
			total += (x&1);
			x >>= 1;
		}

		if(total == 1) 
			return 1; 
		else 
			return 0;
	}
}

Probelam II: Remove Vowels

Solution

public class Solution{
	public String removeVowel(String str) {
		if(str == null || str.length == 0)
			return str;
		StringBuffer sb = new StringBuffer();
		String dict = "aeiouAEIOU";
		for(int i = 0; i < str.length(); i++){
			if(dict.indexOf(str.charAt(i)) > -1)
		 		continue;
			sb.append(str.charAt(i));
		}
		return sb.toStirng();
	}
}

Problem III:

String rotation (if string is the rotate of the other)

Solution

public class Solution{
	public boolean isRotation(String s1,String s2) {  
	    if(s1 == null || s2 == null)
      		return false;
    	if(s1.length() != s2.length())
      		return false;
    	return (s1 + s1).indexOf(s2) != -1;
	}
}
